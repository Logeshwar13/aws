FIRESTORE RULES (Temporary - Allow all writes)
===========================================

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /{document=**} {
      allow read, write: if true;
    }
  }
}

===========================================
STORAGE RULES (Temporary - Allow all writes)
===========================================

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    match /{allPaths=**} {
      allow read, write: if true;
    }
  }
}
